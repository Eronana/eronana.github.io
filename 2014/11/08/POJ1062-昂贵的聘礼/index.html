POJ1062 昂贵的聘礼

这题很容易就能看出是最短路径问题
把交易的人看成点,通过交换减少花费看成边,然后建图
不过难点在于等级的限制
一开始是以酋长为基准,读入数据的时候直接过滤掉超范围的
接着开始反思,在扩点的时候判断...
WA..WA..WA
卡了好久之后百度别人的题解,然后豁然开朗
这题需要枚举区间,感觉就像是迭代加深搜素一样
每次把level范围限定在某个区间内进行Dijkstra,当然,这个区间长度为m
这样在m范围内的交易都可以进行
每次Dijkstra后,把dist扫一遍,找出距离+点值最小的
完事之后直接输出答案

写Dijkstra的时候还脑残了,大概是略久没写吧...松弛都写错了orz

下面是AC代码(824K,16MS)

#define LOCAL
#include <erona>

const int N=101;
int head[N*N];
int money[N];
int dist[N];
int level[N];
struct Edge
{
    int v,w;
    int next;
}edge[N*N];
int cnt=0;
void addEdge(int u,int v,int w)
{
    edge[cnt].v=v;
    edge[cnt].w=w;
    edge[cnt].next=head[u];
    head[u]=cnt++;
}
struct Point
{
    int dist,x;
    bool operator<(const Point &p) const
    {
        return dist>p.dist;
    }
};
void Main()
{
    RDI(m,n);
    REPE(n)head[i]=-1;
    REPE(n)
    {
        RD(money[i],level[i]);
        RDI(x);
        while(x--)
        {
            RDI(t,v);
            addEdge(i,t,v);
        }
    }
    int ans=INF;
    FORE(i,level[1]-m,level[1])
    {
        priority_queue<Point> q;
        q.push({money[1],1});
        CLR(dist,INF);
        dist[1]=0;
        int l=i,r=i+m;
        while(q.size())
        {
            Point p=q.top();q.pop();
            for(int x=head[p.x];~x;x=edge[x].next)
            {
                if(level[edge[x].v]<l||level[edge[x].v]>r)continue;
                int ndist=dist[p.x]+edge[x].w;
                if(dist[edge[x].v]>ndist)q.push({dist[edge[x].v]=ndist,edge[x].v});
            }
        }
        REPE(n)ans=min(ans,dist[i]+money[i]);
    }
    WL(ans);
}

//并查集
struct Disjointset
{
    INT *Father;
    INT n;
    Disjointset(INT _n):n(_n+1)
    {
        Father=new INT[n];
        Init();
    }
    ~Disjointset()
    {
        delete Father;
    }
    void Init()
    {
        for(INT i=1;i<=n;i++)Father[i]=i;
    }
    INT Find(INT a)
    {
        return Father[a]==a?a:Father[a]=Find(Father[a]);
    }
    bool Union(int a,int b)
    {
        INT x=Find(a);
        INT y=Find(b);
        if(x==y)return false;
        Father[x]=y;
        return true;
    }
};
//eraseone(multiset<T>,T)
template<typename T>
bool eraseone(T &c,typename T::value_type v)
{
    typename T::iterator iter(c.find(v));
    if(iter==c.end())return false;
    c.erase(iter);
    return true;
}
